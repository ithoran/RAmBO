);

CREATE TABLE KONTAKT (
    KORISNIK_ID int,
    KONTAKT varchar(30),
    FOREIGN KEY (KORISNIK_ID)
    REFERENCES KORISNIK(ID)
    ON DELETE CASCADE,

);
